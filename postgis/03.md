  google.maps.event.addListener(map, 'idle', idleMaps);

function idleMaps()
{
  var zoom = map.getZoom();
  var bounds = map.getBounds();
  var ne = bounds.getNorthEast();
  var sw = bounds.getSouthWest();

    type: 'GET',
    dataType: 'json'
  })

  })

}
