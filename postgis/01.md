    var map = null;

    function initMap() {
        map = new google.maps.Map($('#map').get(0), {
            center: {lat: 35.172899, lng: 136.887531},
            zoom: 15
            });

        $.ajax({
            url: 'api.php',
            type: 'GET',
            dataType: 'json'
        }).done(function(data){
            console.log('done');
            for(var i in data){
                var circle = new google.maps.Circle({
                    map: map,
                    center: {lat: data[i].coordinates[1],lng:data[i].coordinates[0]},
                    radius: 100
                });
            }
        }).fail(function(){
            console.log('error');
        });
    }

api.php

    <?php

    define('DSN', 'dbname=opendata user=guest');

    $db = pg_connect(DSN);
    if(!$db){
        echo 'error not connected';
        return;
    }

    $sql = <<<EOS
    SELECT ST_AsGeoJson((geom)) AS geojson
    FROM mcdnald2
    EOS;

    $result = pg_prepare($db, '', $sql);
    $result = pg_execute($db, '', []);

    $json = [];
    while(($row = pg_fetch_assoc($result)) != NULL){
        $json[] = json_decode($row['geojson']);
    }

    echo json_encode($json);
